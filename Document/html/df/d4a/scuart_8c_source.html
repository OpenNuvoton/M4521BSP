113uint32_t SCUART_Open(SC_T* sc, uint32_t u32baudrate)
114{
115 uint32_t u32Clk = SCUART_GetClock(sc), u32Div;
116
117 // Calculate divider for target baudrate
118 u32Div = (u32Clk + (u32baudrate >> 1) - 1) / u32baudrate - 1;
119
120 sc->CTL = SC_CTL_SCEN_Msk | SC_CTL_NSB_Msk; // Enable smartcard interface and stop bit = 1
121 sc->UARTCTL = SCUART_CHAR_LEN_8 | SCUART_PARITY_NONE | SC_UARTCTL_UARTEN_Msk; // Enable UART mode, disable parity and 8 bit per character
122 sc->ETUCTL = u32Div;
123
124 return(u32Clk / (u32Div+1));

218uint32_t SCUART_Write(SC_T* sc, uint8_t *pu8TxBuf, uint32_t u32WriteBytes)
219{
220 uint32_t u32Count;
221 /* Baudrate * (start bit + 8-bit data + 1-bit parity + 2-bit stop) */
222 uint32_t u32Delay = (SystemCoreClock / SCUART_GetClock(sc)) * sc->ETUCTL * 12, i;
223
224 for(u32Count = 0; u32Count != u32WriteBytes; u32Count++)
225 {
226 while(SCUART_GET_TX_FULL(sc)) // Wait 'til FIFO not full
227 {
228 /* Block longer than expected. Maybe some interrupt disable SCUART clock? */
229 if(i++ > u32Delay)
230 {
231 g_SCUART_i32ErrCode = SCUART_TIMEOUT_ERR;
232 return u32Count;
233 }
234 }
235 sc->DAT = pu8TxBuf[u32Count]; // Write 1 byte to FIFO
236 }
237 return u32Count;
