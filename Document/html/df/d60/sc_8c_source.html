41uint32_t SC_IsCardInserted(SC_T *sc)
42{
43 // put conditions into two variable to remove IAR compilation warning
44 uint32_t cond1 = ((sc->STATUS & SC_STATUS_CDPINSTS_Msk) >> SC_STATUS_CDPINSTS_Pos);
45 uint32_t cond2 = ((sc->CTL & SC_CTL_CDLV_Msk) >> SC_CTL_CDLV_Pos);
46
47 if(sc == SC0 && u32CardStateIgnore[0] == 1)
48 return TRUE;
49#if 0 /* M4521 has only one SC interface */
50 else if(sc == SC1 && u32CardStateIgnore[1] == 1)
51 return TRUE;
52 else if(sc == SC2 && u32CardStateIgnore[2] == 1)
53 return TRUE;
54 else if(sc == SC3 && u32CardStateIgnore[3] == 1)
55 return TRUE;
56 else if(sc == SC4 && u32CardStateIgnore[4] == 1)
57 return TRUE;
58 else if(sc == SC5 && u32CardStateIgnore[5] == 1)
59 return TRUE;
60#endif
61 else if(cond1 != cond2)
62 return FALSE;
63 else
64 return TRUE;
65}

105void SC_Open(SC_T *sc, uint32_t u32CD, uint32_t u32PWR)
106{
107 uint32_t u32Reg = 0, u32Intf;
108 uint32_t u32Timeout;
109
110 if(sc == SC0)
111 u32Intf = 0;
112#if 0 /* M4521 has only one SC interface */
113 else if(sc == SC1)
114 u32Intf = 1;
115 else if(sc == SC2)
116 u32Intf = 2;
117 else if(sc == SC3)
118 u32Intf = 3;
119 else if(sc == SC4)
120 u32Intf = 4;
121 else if(sc == SC5)
122 u32Intf = 5;
123#endif
124 else
125 return ;
126
127 if(u32CD != SC_PIN_STATE_IGNORE)
128 {
129 u32Reg = u32CD ? 0: SC_CTL_CDLV_Msk;
130 u32CardStateIgnore[u32Intf] = 0;
131 }
132 else
133 {
134 u32CardStateIgnore[u32Intf] = 1;
135 }
136
137 g_SC_i32ErrCode = 0;
138 u32Timeout = SystemCoreClock;
139 while ((sc->PINCTL & SC_PINCTL_SYNC_Msk) && (u32Timeout-- > 0));
140 if (sc->PINCTL & SC_PINCTL_SYNC_Msk)
141 g_SC_i32ErrCode = -1;
142 sc->PINCTL = u32PWR ? 0 : SC_PINCTL_PWRINV_Msk;
143
144 u32Timeout = SystemCoreClock;
145 while ((sc->CTL & SC_CTL_SYNC_Msk) && (u32Timeout-- > 0));
146 if (sc->CTL & SC_CTL_SYNC_Msk)
147 g_SC_i32ErrCode = -1;
148 sc->CTL = SC_CTL_SCEN_Msk | u32Reg;
149}

157void SC_ResetReader(SC_T *sc)
158{
159 uint32_t u32Intf;
160 uint32_t u32Timeout;
161
162 if(sc == SC0)
163 u32Intf = 0;
164#if 0 /* M4521 has only one SC interface */
165 else if(sc == SC1)
166 u32Intf = 1;
167 else if(sc == SC2)
168 u32Intf = 2;
169 else if(sc == SC3)
170 u32Intf = 3;
171 else if(sc == SC4)
172 u32Intf = 4;
173 else if(sc == SC5)
174 u32Intf = 5;
175#endif
176 else
177 return ;
178
179 // Reset FIFO, enable auto de-activation while card removal
180 sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk | SC_ALTCTL_ADACEN_Msk);
181 // Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry)
182 u32Timeout = SystemCoreClock;
183 while ((sc->CTL & SC_CTL_SYNC_Msk) && (u32Timeout-- > 0));
184 if (sc->CTL & SC_CTL_SYNC_Msk)
185 g_SC_i32ErrCode = -1;
186
187 sc->CTL &= ~(SC_CTL_RXTRGLV_Msk | SC_CTL_CDDBSEL_Msk | SC_CTL_TXRTY_Msk | SC_CTL_RXRTY_Msk);
188 // Enable auto convention, and all three smartcard internal timers
189 sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
190 // Disable Rx timeout
191 sc->RXTOUT = 0;
192 // 372 clocks per ETU by default
193 sc->ETUCTL = 371;
194
195 /* Enable necessary interrupt for smartcard operation */
196 if(u32CardStateIgnore[u32Intf]) // Do not enable card detect interrupt if card present state ignore
197 sc->INTEN = (SC_INTEN_RDAIEN_Msk |
198 SC_INTEN_TERRIEN_Msk |
199 SC_INTEN_TMR0IEN_Msk |
200 SC_INTEN_TMR1IEN_Msk |
201 SC_INTEN_TMR2IEN_Msk |
202 SC_INTEN_BGTIEN_Msk |
203 SC_INTEN_ACERRIEN_Msk);
204 else
205 sc->INTEN = (SC_INTEN_RDAIEN_Msk |
206 SC_INTEN_TERRIEN_Msk |
207 SC_INTEN_TMR0IEN_Msk |
208 SC_INTEN_TMR1IEN_Msk |
209 SC_INTEN_TMR2IEN_Msk |
210 SC_INTEN_BGTIEN_Msk |
211 SC_INTEN_ACERRIEN_Msk |
212 SC_INTEN_CDIEN_Msk);
213
214 return;
215}
