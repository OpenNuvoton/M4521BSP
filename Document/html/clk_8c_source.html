202 uint32_t CLK_SetCoreClock(uint32_t u32Hclk)
203 {
204  uint32_t u32HIRCSTB;
205 
206  /* Read HIRC clock source stable flag */
207  u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
208 
209  /* The range of u32Hclk is 25 MHz ~ 72 MHz */
210  if(u32Hclk > FREQ_72MHZ)
211  u32Hclk = FREQ_72MHZ;
212  if(u32Hclk < FREQ_25MHZ)
213  u32Hclk = FREQ_25MHZ;
214 
215  /* Switch HCLK clock source to HIRC clock for safe */
216  CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
217  CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
218  CLK->CLKSEL0 |= CLK_CLKSEL0_HCLKSEL_Msk;
219  CLK->CLKDIV0 &= (~CLK_CLKDIV0_HCLKDIV_Msk);
220 
221  /* Configure PLL setting if HXT clock is stable */
222  if(CLK->STATUS & CLK_STATUS_HXTSTB_Msk)
223  u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HXT, (u32Hclk << 1));
224 
225  /* Configure PLL setting if HXT clock is not stable */
226  else
227  {
228  u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HIRC, (u32Hclk << 1));
229 
230  /* Read HIRC clock source stable flag */
231  u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
232  }
233 
234  /* Select HCLK clock source to PLL,
235  Select HCLK clock source divider as 2
236  and update system core clock
237  */
238  CLK_SetHCLK(CLK_CLKSEL0_HCLKSEL_PLL, CLK_CLKDIV0_HCLK(2));
239 
240  /* Disable HIRC if HIRC is disabled before setting core clock */
241  if(u32HIRCSTB == 0)
242  CLK->PWRCTL &= ~CLK_PWRCTL_HIRCEN_Msk;
243 
244  /* Return actually HCLK frequency is PLL frequency divide 2 */
245  return u32Hclk >> 1;
246 }

263 {

268

523 uint32_t CLK_EnablePLL(uint32_t u32PllClkSrc, uint32_t u32PllFreq)
524 {
525  uint32_t u32PllSrcClk, u32NR, u32NF, u32NO, u32CLK_SRC;
526  uint32_t u32Tmp, u32Tmp2, u32Tmp3, u32Min, u32MinNF, u32MinNR;
527 
528  /* Disable PLL first to avoid unstable when setting PLL */
529  CLK_DisablePLL();
530 
531  /* PLL source clock is from HXT */
532  if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
533  {
534  /* Enable HXT clock */
535  CLK->PWRCTL |= CLK_PWRCTL_HXTEN_Msk;
536 
537  /* Wait for HXT clock ready */
538  CLK_WaitClockReady(CLK_STATUS_HXTSTB_Msk);
539 
540  /* Select PLL source clock from HXT */
541  u32CLK_SRC = CLK_PLLCTL_PLLSRC_HXT;
542  u32PllSrcClk = __HXT;
543 
544  /* u32NR start from 2 */
545  u32NR = 2;
546  }
547 
548  /* PLL source clock is from HIRC */
549  else
550  {
551  /* Enable HIRC clock */
552  CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
553 
554  /* Wait for HIRC clock ready */
555  CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
556 
557  /* Select PLL source clock from HIRC */
558  u32CLK_SRC = CLK_PLLCTL_PLLSRC_HIRC;
559  u32PllSrcClk = __HIRC;
560 
561  /* u32NR start from 4 when FIN = 22.1184MHz to avoid calculation overflow */
562  u32NR = 4;
563  }
564 
565  /* Select "NO" according to request frequency */
566  if((u32PllFreq <= FREQ_500MHZ) && (u32PllFreq > FREQ_250MHZ))
567  {
568  u32NO = 0;
569  }
570  else if((u32PllFreq <= FREQ_250MHZ) && (u32PllFreq > FREQ_125MHZ))
571  {
572  u32NO = 1;
573  u32PllFreq = u32PllFreq << 1;
574  }
575  else if((u32PllFreq <= FREQ_125MHZ) && (u32PllFreq >= FREQ_50MHZ))
576  {
577  u32NO = 3;
578  u32PllFreq = u32PllFreq << 2;
579  }
580  else
581  {
582  /* Wrong frequency request. Just return default setting. */
583  goto lexit;
584  }
585 
586  /* Find best solution */
587  u32Min = (uint32_t) - 1;
588  u32MinNR = 0;
589  u32MinNF = 0;
590  for(; u32NR <= 33; u32NR++)
591  {
592  u32Tmp = u32PllSrcClk / u32NR;
593  if((u32Tmp > 1600000) && (u32Tmp < 16000000))
594  {
595  for(u32NF = 2; u32NF <= 513; u32NF++)
596  {
597  u32Tmp2 = u32Tmp * u32NF;
598  if((u32Tmp2 >= 200000000) && (u32Tmp2 <= 500000000))
599  {
600  u32Tmp3 = (u32Tmp2 > u32PllFreq) ? u32Tmp2 - u32PllFreq : u32PllFreq - u32Tmp2;
601  if(u32Tmp3 < u32Min)
602  {
603  u32Min = u32Tmp3;
604  u32MinNR = u32NR;
605  u32MinNF = u32NF;
606 
607  /* Break when get good results */
608  if(u32Min == 0)
609  break;
610  }
611  }
612  }
613  }
614  }
615 
616  /* Enable and apply new PLL setting. */
617  CLK->PLLCTL = u32CLK_SRC | (u32NO << 14) | ((u32MinNR - 2) << 9) | (u32MinNF - 2);
618 
619  /* Wait for PLL clock stable */
620  CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
621 
622  /* Return actual PLL output clock frequency */
623  return u32PllSrcClk / ((u32NO + 1) * u32MinNR) * u32MinNF;
624 
625 lexit:
626 
627  /* Apply default PLL setting and return */
628  if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
629  CLK->PLLCTL = CLK_PLLCTL_72MHz_HXT; /* 72MHz */
630  else
631  CLK->PLLCTL = CLK_PLLCTL_72MHz_HIRC; /* 71.8848MHz */
632 
633  /* Wait for PLL clock stable */
634  CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
635 
636  return CLK_GetPLLClockFreq();
637
