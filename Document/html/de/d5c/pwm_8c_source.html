38uint32_t PWM_ConfigCaptureChannel(PWM_T *pwm, uint32_t u32ChannelNum, uint32_t u32UnitTimeNsec, uint32_t u32CaptureEdge)
39{
40 uint32_t u32Src;
41 uint32_t u32PWMClockSrc;
42 uint32_t u32NearestUnitTimeNsec;
43 uint16_t u16Prescale = 1, u16CNR = 0xFFFF;
44
45 if(pwm == PWM0)
46 u32Src = CLK->CLKSEL2 & CLK_CLKSEL2_PWM0SEL_Msk;
47 else//(pwm == PWM1)
48 u32Src = CLK->CLKSEL2 & CLK_CLKSEL2_PWM1SEL_Msk;
49
50 if(u32Src == 0)
51 {
52 //clock source is from PLL clock
53 u32PWMClockSrc = CLK_GetPLLClockFreq();
54 }
55 else
56 {
57 //clock source is from PCLK
58 SystemCoreClockUpdate();
59 u32PWMClockSrc = SystemCoreClock;
60 }
61
62 u32PWMClockSrc /= 1000;
63 for(u16Prescale = 1; u16Prescale <= 0x1000; u16Prescale++)
64 {
65 u32NearestUnitTimeNsec = (1000000 * u16Prescale) / u32PWMClockSrc;
66 if(u32NearestUnitTimeNsec < u32UnitTimeNsec)
67 {
68 if(u16Prescale == 0x1000) //limit to the maximum unit time(nano second)
69 break;
70 if(!((1000000 * (u16Prescale + 1) > (u32NearestUnitTimeNsec * u32PWMClockSrc))))
71 break;
72 continue;
73 }
74 break;
75 }
76
77 // convert to real register value
78 // every two channels share a prescaler
79 PWM_SET_PRESCALER(pwm, u32ChannelNum, --u16Prescale);
80
81 // set PWM to down count type(edge aligned)
82 (pwm)->CTL1 = ((pwm)->CTL1 & ~(PWM_CTL1_CNTTYPE0_Msk << (2 * u32ChannelNum))) | (1UL << (2 * u32ChannelNum));
83 // set PWM to auto-reload mode
84 (pwm)->CTL1 &= ~(PWM_CTL1_CNTMODE0_Msk << u32ChannelNum);
85 PWM_SET_CNR(pwm, u32ChannelNum, u16CNR);
86
87 return (u32NearestUnitTimeNsec);
88}

102uint32_t PWM_ConfigOutputChannel(PWM_T *pwm, uint32_t u32ChannelNum, uint32_t u32Frequency, uint32_t u32DutyCycle)
103{
104 uint32_t u32Src;
105 uint32_t u32PWMClockSrc;
106 uint32_t i;
107 uint16_t u16Prescale = 1, u16CNR = 0xFFFF;
108
109 if(pwm == PWM0)
110 u32Src = CLK->CLKSEL2 & CLK_CLKSEL2_PWM0SEL_Msk;
111 else//(pwm == PWM1)
112 u32Src = CLK->CLKSEL2 & CLK_CLKSEL2_PWM1SEL_Msk;
113
114 if(u32Src == 0)
115 {
116 //clock source is from PLL clock
117 u32PWMClockSrc = CLK_GetPLLClockFreq();
118 }
119 else
120 {
121 //clock source is from PCLK
122 SystemCoreClockUpdate();
123 u32PWMClockSrc = SystemCoreClock;
124 }
125
126 for(u16Prescale = 1; u16Prescale < 0xFFF; u16Prescale++) //prescale could be 0~0xFFF
127 {
128 i = (u32PWMClockSrc / u32Frequency) / u16Prescale;
129 // If target value is larger than CNR, need to use a larger prescaler
130 if(i > (0x10000))
131 continue;
132
133 u16CNR = i;
134 break;
135 }
136 // Store return value here 'cos we're gonna change u16Prescale & u16CNR to the real value to fill into register
137 i = u32PWMClockSrc / (u16Prescale * u16CNR);
138
139 // convert to real register value
140 // every two channels share a prescaler
141 PWM_SET_PRESCALER(pwm, u32ChannelNum, --u16Prescale);
142 // set PWM to down count type(edge aligned)
143 (pwm)->CTL1 = ((pwm)->CTL1 & ~(PWM_CTL1_CNTTYPE0_Msk << (2 * u32ChannelNum))) | (1UL << (2 * u32ChannelNum));
144 // set PWM to auto-reload mode
145 (pwm)->CTL1 &= ~(PWM_CTL1_CNTMODE0_Msk << u32ChannelNum);
146
147 PWM_SET_CNR(pwm, u32ChannelNum, --u16CNR);
148 if(u32DutyCycle)
149 {
150 PWM_SET_CMR(pwm, u32ChannelNum, u32DutyCycle * (u16CNR + 1) / 100 - 1);
151 (pwm)->WGCTL0 &= ~((PWM_WGCTL0_PRDPCTL0_Msk | PWM_WGCTL0_ZPCTL0_Msk) << (u32ChannelNum * 2));
152 (pwm)->WGCTL0 |= (PWM_OUTPUT_LOW << (u32ChannelNum * 2 + PWM_WGCTL0_PRDPCTL0_Pos));
153 (pwm)->WGCTL1 &= ~((PWM_WGCTL1_CMPDCTL0_Msk | PWM_WGCTL1_CMPUCTL0_Msk) << (u32ChannelNum * 2));
154 (pwm)->WGCTL1 |= (PWM_OUTPUT_HIGH << (u32ChannelNum * 2 + PWM_WGCTL1_CMPDCTL0_Pos));
155 }
156 else
157 {
158 PWM_SET_CMR(pwm, u32ChannelNum, 0);
159 (pwm)->WGCTL0 &= ~((PWM_WGCTL0_PRDPCTL0_Msk | PWM_WGCTL0_ZPCTL0_Msk) << (u32ChannelNum * 2));
160 (pwm)->WGCTL0 |= (PWM_OUTPUT_LOW << (u32ChannelNum * 2 + PWM_WGCTL0_ZPCTL0_Pos));
161 (pwm)->WGCTL1 &= ~((PWM_WGCTL1_CMPDCTL0_Msk | PWM_WGCTL1_CMPUCTL0_Msk) << (u32ChannelNum * 2));
162 (pwm)->WGCTL1 |= (PWM_OUTPUT_HIGH << (u32ChannelNum * 2 + PWM_WGCTL1_CMPDCTL0_Pos));
163 }
164
165 return(i);
166}
