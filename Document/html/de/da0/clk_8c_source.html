204uint32_t CLK_SetCoreClock(uint32_t u32Hclk)
205{
206 uint32_t u32HIRCSTB;
207
208 /* Read HIRC clock source stable flag */
209 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
210
211 /* The range of u32Hclk is 25 MHz ~ 72 MHz */
212 if(u32Hclk > FREQ_72MHZ)
213 u32Hclk = FREQ_72MHZ;
214 if(u32Hclk < FREQ_25MHZ)
215 u32Hclk = FREQ_25MHZ;
216
217 /* Switch HCLK clock source to HIRC clock for safe */
218 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
219 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
220 CLK->CLKSEL0 |= CLK_CLKSEL0_HCLKSEL_Msk;
221 CLK->CLKDIV0 &= (~CLK_CLKDIV0_HCLKDIV_Msk);
222
223 /* Configure PLL setting if HXT clock is stable */
224 if(CLK->STATUS & CLK_STATUS_HXTSTB_Msk)
225 u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HXT, (u32Hclk << 1));
226
227 /* Configure PLL setting if HXT clock is not stable */
228 else
229 {
230 u32Hclk = CLK_EnablePLL(CLK_PLLCTL_PLLSRC_HIRC, (u32Hclk << 1));
231
232 /* Read HIRC clock source stable flag */
233 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
234 }
235
236 /* Select HCLK clock source to PLL,
237 Select HCLK clock source divider as 2
238 and update system core clock
239 */
240 CLK_SetHCLK(CLK_CLKSEL0_HCLKSEL_PLL, CLK_CLKDIV0_HCLK(2));
241
242 /* Disable HIRC if HIRC is disabled before setting core clock */
243 if(u32HIRCSTB == 0)
244 CLK->PWRCTL &= ~CLK_PWRCTL_HIRCEN_Msk;
245
246 /* Return actually HCLK frequency is PLL frequency divide 2 */
247 return u32Hclk >> 1;
248}

264void CLK_SetHCLK(uint32_t u32ClkSrc, uint32_t u32ClkDiv)
265{
266 uint32_t u32HIRCSTB;
267
268 /* Read HIRC clock source stable flag */
269 u32HIRCSTB = CLK->STATUS & CLK_STATUS_HIRCSTB_Msk;
270
271 /* Switch to HIRC for Safe. Avoid HCLK too high when applying new divider. */
272 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
273 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
274 CLK->CLKSEL0 = (CLK->CLKSEL0 & (~CLK_CLKSEL0_HCLKSEL_Msk)) | CLK_CLKSEL0_HCLKSEL_HIRC;
275
276 /* Apply new Divider */
277 CLK->CLKDIV0 = (CLK->CLKDIV0 & (~CLK_CLKDIV0_HCLKDIV_Msk)) | u32ClkDiv;
278
279 /* Switch HCLK to new HCLK source */
280 CLK->CLKSEL0 = (CLK->CLKSEL0 & (~CLK_CLKSEL0_HCLKSEL_Msk)) | u32ClkSrc;
281
282 /* Update System Core Clock */
283 SystemCoreClockUpdate();
284
285 /* Disable HIRC if HIRC is disabled before switching HCLK source */
286 if(u32HIRCSTB == 0)
287 CLK->PWRCTL &= ~CLK_PWRCTL_HIRCEN_Msk;
288}

525uint32_t CLK_EnablePLL(uint32_t u32PllClkSrc, uint32_t u32PllFreq)
526{
527 uint32_t u32PllSrcClk, u32NR, u32NF, u32NO, u32CLK_SRC;
528 uint32_t u32Tmp, u32Tmp2, u32Tmp3, u32Min, u32MinNF, u32MinNR;
529
530 /* Disable PLL first to avoid unstable when setting PLL */
531 CLK_DisablePLL();
532
533 /* PLL source clock is from HXT */
534 if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
535 {
536 /* Enable HXT clock */
537 CLK->PWRCTL |= CLK_PWRCTL_HXTEN_Msk;
538
539 /* Wait for HXT clock ready */
540 CLK_WaitClockReady(CLK_STATUS_HXTSTB_Msk);
541
542 /* Select PLL source clock from HXT */
543 u32CLK_SRC = CLK_PLLCTL_PLLSRC_HXT;
544 u32PllSrcClk = __HXT;
545
546 /* u32NR start from 2 */
547 u32NR = 2;
548 }
549
550 /* PLL source clock is from HIRC */
551 else
552 {
553 /* Enable HIRC clock */
554 CLK->PWRCTL |= CLK_PWRCTL_HIRCEN_Msk;
555
556 /* Wait for HIRC clock ready */
557 CLK_WaitClockReady(CLK_STATUS_HIRCSTB_Msk);
558
559 /* Select PLL source clock from HIRC */
560 u32CLK_SRC = CLK_PLLCTL_PLLSRC_HIRC;
561 u32PllSrcClk = __HIRC;
562
563 /* u32NR start from 4 when FIN = 22.1184MHz to avoid calculation overflow */
564 u32NR = 4;
565 }
566
567 /* Select "NO" according to request frequency */
568 if((u32PllFreq <= FREQ_500MHZ) && (u32PllFreq > FREQ_250MHZ))
569 {
570 u32NO = 0;
571 }
572 else if((u32PllFreq <= FREQ_250MHZ) && (u32PllFreq > FREQ_125MHZ))
573 {
574 u32NO = 1;
575 u32PllFreq = u32PllFreq << 1;
576 }
577 else if((u32PllFreq <= FREQ_125MHZ) && (u32PllFreq >= FREQ_50MHZ))
578 {
579 u32NO = 3;
580 u32PllFreq = u32PllFreq << 2;
581 }
582 else
583 {
584 /* Wrong frequency request. Just return default setting. */
585 goto lexit;
586 }
587
588 /* Find best solution */
589 u32Min = (uint32_t) - 1;
590 u32MinNR = 0;
591 u32MinNF = 0;
592 for(; u32NR <= 33; u32NR++)
593 {
594 u32Tmp = u32PllSrcClk / u32NR;
595 if((u32Tmp > 1600000) && (u32Tmp < 16000000))
596 {
597 for(u32NF = 2; u32NF <= 513; u32NF++)
598 {
599 u32Tmp2 = u32Tmp * u32NF;
600 if((u32Tmp2 >= 200000000) && (u32Tmp2 <= 500000000))
601 {
602 u32Tmp3 = (u32Tmp2 > u32PllFreq) ? u32Tmp2 - u32PllFreq : u32PllFreq - u32Tmp2;
603 if(u32Tmp3 < u32Min)
604 {
605 u32Min = u32Tmp3;
606 u32MinNR = u32NR;
607 u32MinNF = u32NF;
608
609 /* Break when get good results */
610 if(u32Min == 0)
611 break;
612 }
613 }
614 }
615 }
616 }
617
618 /* Enable and apply new PLL setting. */
619 CLK->PLLCTL = u32CLK_SRC | (u32NO << 14) | ((u32MinNR - 2) << 9) | (u32MinNF - 2);
620
621 /* Wait for PLL clock stable */
622 CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
623
624 /* Return actual PLL output clock frequency */
625 return u32PllSrcClk / ((u32NO + 1) * u32MinNR) * u32MinNF;
626
627lexit:
628
629 /* Apply default PLL setting and return */
630 if(u32PllClkSrc == CLK_PLLCTL_PLLSRC_HXT)
631 CLK->PLLCTL = CLK_PLLCTL_72MHz_HXT; /* 72MHz */
632 else
633 CLK->PLLCTL = CLK_PLLCTL_72MHz_HIRC; /* 71.8848MHz */
634
635 /* Wait for PLL clock stable */
636 CLK_WaitClockReady(CLK_STATUS_PLLSTB_Msk);
637
638 return CLK_GetPLLClockFreq();
639
640}

669uint32_t CLK_WaitClockReady(uint32_t u32ClkMask)
670{
671 uint32_t u32TimeOutCnt = SystemCoreClock / 2;
672 uint32_t u32Ret = 1U;
673
674 g_CLK_i32ErrCode = 0;
675
676 while((CLK->STATUS & u32ClkMask) != u32ClkMask)
677 {
678 if(--u32TimeOutCnt == 0)
679 {
680 u32Ret = 0U;
681 break;
682 }
683 }
684
685 if(u32TimeOutCnt == 0)
686 g_CLK_i32ErrCode = CLK_TIMEOUT_ERR;
687
