145 void SC_ResetReader(SC_T *sc)
146 {
147  uint32_t u32Intf;
148 
149  if(sc == SC0)
150  u32Intf = 0;
151 #if 0 /* M4521 has only one SC interface */
152  else if(sc == SC1)
153  u32Intf = 1;
154  else if(sc == SC2)
155  u32Intf = 2;
156  else if(sc == SC3)
157  u32Intf = 3;
158  else if(sc == SC4)
159  u32Intf = 4;
160  else if(sc == SC5)
161  u32Intf = 5;
162 #endif
163  else
164  return ;
165 
166  // Reset FIFO, enable auto de-activation while card removal
167  sc->ALTCTL |= (SC_ALTCTL_TXRST_Msk | SC_ALTCTL_RXRST_Msk | SC_ALTCTL_ADACEN_Msk);
168  // Set Rx trigger level to 1 character, longest card detect debounce period, disable error retry (EMV ATR does not use error retry)
169  while(sc->CTL & SC_CTL_SYNC_Msk);
170  sc->CTL &= ~(SC_CTL_RXTRGLV_Msk | SC_CTL_CDDBSEL_Msk | SC_CTL_TXRTY_Msk | SC_CTL_RXRTY_Msk);
171  // Enable auto convention, and all three smartcard internal timers
172  sc->CTL |= SC_CTL_AUTOCEN_Msk | SC_CTL_TMRSEL_Msk;
173  // Disable Rx timeout
174  sc->RXTOUT = 0;
175  // 372 clocks per ETU by default
176  sc->ETUCTL = 371;
177 
178  /* Enable necessary interrupt for smartcard operation */
179  if(u32CardStateIgnore[u32Intf]) // Do not enable card detect interrupt if card present state ignore
180  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
181  SC_INTEN_TERRIEN_Msk |
182  SC_INTEN_TMR0IEN_Msk |
183  SC_INTEN_TMR1IEN_Msk |
184  SC_INTEN_TMR2IEN_Msk |
185  SC_INTEN_BGTIEN_Msk |
186  SC_INTEN_ACERRIEN_Msk);
187  else
188  sc->INTEN = (SC_INTEN_RDAIEN_Msk |
189  SC_INTEN_TERRIEN_Msk |
190  SC_INTEN_TMR0IEN_Msk |
191  SC_INTEN_TMR1IEN_Msk |
192  SC_INTEN_TMR2IEN_Msk |
193  SC_INTEN_BGTIEN_Msk |
194  SC_INTEN_ACERRIEN_Msk |
195  SC_INTEN_CDIEN_Msk);
196 
197  return;
198 }
