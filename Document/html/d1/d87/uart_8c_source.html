44void UART_ClearIntFlag(UART_T* uart, uint32_t u32InterruptFlag)
45{
46
47 if(u32InterruptFlag & UART_INTSTS_RLSINT_Msk) /* Clear Receive Line Status Interrupt */
48 {
49 uart->FIFOSTS = UART_FIFOSTS_BIF_Msk | UART_FIFOSTS_FEF_Msk | UART_FIFOSTS_PEF_Msk;
50 uart->FIFOSTS = UART_FIFOSTS_ADDRDETF_Msk;
51 }
52
53 if(u32InterruptFlag & UART_INTSTS_MODEMINT_Msk) /* Clear Modem Status Interrupt */
54 uart->MODEMSTS |= UART_MODEMSTS_CTSDETF_Msk;
55
56 if(u32InterruptFlag & UART_INTSTS_BUFERRINT_Msk) /* Clear Buffer Error Interrupt */
57 {
58 uart->FIFOSTS = UART_FIFOSTS_RXOVIF_Msk | UART_FIFOSTS_TXOVIF_Msk;
59 }
60
61 if(u32InterruptFlag & UART_INTSTS_CTSWKIF_Msk) /* Clear CTS Wake-up Interrupt */
62 {
63 uart->INTSTS = UART_INTSTS_CTSWKIF_Msk;
64 }
65
66 if(u32InterruptFlag & UART_INTSTS_DATWKIF_Msk) /* Clear Data Wake-up Interrupt */
67 {
68 uart->INTSTS = UART_INTSTS_DATWKIF_Msk;
69 }
70}

185void UART_Open(UART_T* uart, uint32_t u32baudrate)
186{
187 uint8_t u8UartClkSrcSel, u8UartClkDivNum;
188 uint32_t u32ClkTbl[4] = {__HXT, 0, __LXT, __HIRC};
189 uint32_t u32Baud_Div = 0;
190
191 /* Get UART clock source selection */
192 u8UartClkSrcSel = (CLK->CLKSEL1 & CLK_CLKSEL1_UARTSEL_Msk) >> CLK_CLKSEL1_UARTSEL_Pos;
193
194 /* Get UART clock divider number */
195 u8UartClkDivNum = (CLK->CLKDIV0 & CLK_CLKDIV0_UARTDIV_Msk) >> CLK_CLKDIV0_UARTDIV_Pos;
196
197 /* Select UART function */
198 uart->FUNCSEL = UART_FUNCSEL_UART;
199
200 /* Set UART line configuration */
201 uart->LINE = UART_WORD_LEN_8 | UART_PARITY_NONE | UART_STOP_BIT_1;
202
203 /* Set UART Rx and RTS trigger level */
204 uart->FIFO &= ~(UART_FIFO_RFITL_Msk | UART_FIFO_RTSTRGLV_Msk);
205
206 /* Get PLL clock frequency if UART clock source selection is PLL */
207 if(u8UartClkSrcSel == 1)
208 u32ClkTbl[u8UartClkSrcSel] = CLK_GetPLLClockFreq();
209
210 /* Set UART baud rate */
211 if(u32baudrate != 0)
212 {
213 u32Baud_Div = UART_BAUD_MODE2_DIVIDER((u32ClkTbl[u8UartClkSrcSel]) / (u8UartClkDivNum + 1), u32baudrate);
214
215 if(u32Baud_Div > 0xFFFF)
216 uart->BAUD = (UART_BAUD_MODE0 | UART_BAUD_MODE0_DIVIDER((u32ClkTbl[u8UartClkSrcSel]) / (u8UartClkDivNum + 1), u32baudrate));
217 else
218 uart->BAUD = (UART_BAUD_MODE2 | u32Baud_Div);
219 }
220}

234uint32_t UART_Read(UART_T* uart, uint8_t *pu8RxBuf, uint32_t u32ReadBytes)
235{
236 uint32_t u32Count, u32delayno;
237 uint32_t u32Timeout = 3 * SystemCoreClock; // timeout 3 second
238
239 for(u32Count = 0; u32Count < u32ReadBytes; u32Count++)
240 {
241 u32delayno = 0;
242
243 while(uart->FIFOSTS & UART_FIFOSTS_RXEMPTY_Msk) /* Check RX empty => failed */
244 {
245 u32delayno++;
246 if(u32delayno >= u32Timeout)
247 return u32Count;
248 }
249 pu8RxBuf[u32Count] = uart->DAT; /* Get Data from UART RX */
250 }
251
252 return u32Count;

429uint32_t UART_Write(UART_T* uart, uint8_t *pu8TxBuf, uint32_t u32WriteBytes)
430{
431 uint32_t u32Count, u32delayno;
432 uint32_t u32Exit = 0ul;
433 uint32_t u32Timeout = 3 * SystemCoreClock; // timeout 3 second
434
435 for (u32Count = 0ul; u32Count != u32WriteBytes; u32Count++)
436 {
437 u32delayno = 0ul;
438
439 while (uart->FIFOSTS & UART_FIFOSTS_TXFULL_Msk) /* Check Tx Full */
440 {
441 u32delayno++;
442
443 if (u32delayno >= u32Timeout)
444 {
445 u32Exit = 1ul;
446 break;
447 }
448 }
449
450 if (u32Exit == 1ul)
451 {
452 break;
453 }
454 else
455 {
456 uart->DAT = pu8TxBuf[u32Count]; /* Send UART Data from buffer */
457 }
458 }
459
460 return u32Count;
