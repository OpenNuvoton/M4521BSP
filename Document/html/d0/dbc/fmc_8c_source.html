418int32_t FMC_Write256(uint32_t u32Addr, uint32_t *pu32Buf)
419{
420 int32_t i, idx;
421 volatile uint32_t *pu32IspData;
422 int32_t tout;
423
424 g_FMC_i32ErrCode = 0;
425 idx = 0;
426 FMC->ISPCMD = FMC_ISPCMD_MULTI_PROG;
427 FMC->ISPADDR = u32Addr;
428
429retrigger:
430
431 FMC->MPDAT0 = pu32Buf[idx + 0];
432 FMC->MPDAT1 = pu32Buf[idx + 1];
433 FMC->MPDAT2 = pu32Buf[idx + 2];
434 FMC->MPDAT3 = pu32Buf[idx + 3];
435 FMC->ISPTRG = 0x1;
436
437 pu32IspData = &FMC->MPDAT0;
438 idx += 4;
439
440 for(i = idx; i < 256 / 4; i += 4) // Max data length is 256 bytes (256/4 words)
441 {
442 __set_PRIMASK(1); // Mask interrupt to avoid status check coherence error
443 tout = FMC_TIMEOUT_WRITE;
444 do
445 {
446 if ((FMC->MPSTS & FMC_MPSTS_MPBUSY_Msk) == 0)
447 {
448 __set_PRIMASK(0);
449 FMC->ISPADDR = FMC->MPADDR & (~0xful);
450 idx = (FMC->ISPADDR - u32Addr) / 4;
451 goto retrigger;
452 }
453 }
454 while ((tout-- > 0) && (FMC->MPSTS & (3 << FMC_MPSTS_D0_Pos)));
455
456 if (tout <= 0)
457 {
458 g_FMC_i32ErrCode = -1;
459 __set_PRIMASK(0);
460 return -1;
461 }
462
463 tout = FMC_TIMEOUT_WRITE;
464 // Update new data for D0
465 pu32IspData[0] = pu32Buf[i];
466 pu32IspData[1] = pu32Buf[i + 1];
467 do
468 {
469 if ((FMC->MPSTS & FMC_MPSTS_MPBUSY_Msk) == 0)
470 {
471 __set_PRIMASK(0);
472 FMC->ISPADDR = FMC->MPADDR & (~0xful);
473 idx = (FMC->ISPADDR - u32Addr) / 4;
474 goto retrigger;
475 }
476 }
477 while ((tout-- > 0) && (FMC->MPSTS & (3 << FMC_MPSTS_D2_Pos)));
478
479 if (tout <= 0)
480 {
481 g_FMC_i32ErrCode = -1;
482 __set_PRIMASK(0);
483 return -1;
484 }
485
486 // Update new data for D2
487 pu32IspData[2] = pu32Buf[i + 2];
488 pu32IspData[3] = pu32Buf[i + 3];
489 __set_PRIMASK(0);
490 }
491
492 tout = FMC_TIMEOUT_WRITE;
493 while ((tout-- > 0) && (FMC->ISPSTS & FMC_ISPSTS_ISPBUSY_Msk)) { }
494 if (tout <= 0)
495 {
496 g_FMC_i32ErrCode = -1;
497 return -1;
498 }
499 return 0;

602uint32_t FMC_GetCheckSum(uint32_t u32Addr, int32_t i32Size)
603{
604 uint32_t tout;
605
606 g_FMC_i32ErrCode = 0;
607 FMC->ISPCMD = FMC_ISPCMD_CAL_CHECKSUM;
608 FMC->ISPADDR = u32Addr;
609 FMC->ISPDAT = i32Size;
610 FMC->ISPTRG = 0x1;
611#if ISBEN
612 __ISB();
613#endif
614 tout = FMC_TIMEOUT_CHKSUM;
615 while (tout-- > 0)
616 {
617 if (!FMC->ISPTRG) /* Waiting for ISP Done */
618 goto chksum;
619 }
620 g_FMC_i32ErrCode = -1;
621 return 0xFFFFFFFF;
622
623chksum:
624 FMC->ISPCMD = FMC_ISPCMD_CHECKSUM;
625 FMC->ISPTRG = 0x1;
626#if ISBEN
627 __ISB();
628#endif
629 tout = FMC_TIMEOUT_CHKSUM;
630 while (tout-- > 0)
631 {
632 if (!FMC->ISPTRG) /* Waiting for ISP Done */
633 return FMC->ISPDAT;
634 }
635 g_FMC_i32ErrCode = -1;
636 return 0xFFFFFFFF;
