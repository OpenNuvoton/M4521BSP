332__STATIC_INLINE uint32_t CLK_GetPLLClockFreq(void)
333{
334 uint32_t u32PllFreq = 0, u32PllReg;
335 uint32_t u32FIN, u32NF, u32NR, u32NO;
336 uint8_t au8NoTbl[4] = {1, 2, 2, 4};
337
338 u32PllReg = CLK->PLLCTL;
339
340 if(u32PllReg & (CLK_PLLCTL_PD_Msk | CLK_PLLCTL_OE_Msk))
341 return 0; /* PLL is in power down mode or fix low */
342
343 if(u32PllReg & CLK_PLLCTL_PLLSRC_HIRC)
344 u32FIN = __HIRC; /* PLL source clock from HIRC */
345 else
346 u32FIN = __HXT; /* PLL source clock from HXT */
347
348 if(u32PllReg & CLK_PLLCTL_BP_Msk)
349 return u32FIN; /* PLL is in bypass mode */
350
351 /* PLL is output enabled in normal work mode */
352 u32NO = au8NoTbl[((u32PllReg & CLK_PLLCTL_OUTDIV_Msk) >> CLK_PLLCTL_OUTDIV_Pos)];
353 u32NF = ((u32PllReg & CLK_PLLCTL_FBDIV_Msk) >> CLK_PLLCTL_FBDIV_Pos) + 2;
354 u32NR = ((u32PllReg & CLK_PLLCTL_INDIV_Msk) >> CLK_PLLCTL_INDIV_Pos) + 2;
355
356 /* u32FIN is shifted 2 bits to avoid overflow */
357 u32PllFreq = (((u32FIN >> 2) * u32NF) / (u32NR * u32NO) << 2);
358
359 return u32PllFreq;
360}

373__STATIC_INLINE int32_t CLK_SysTickDelay(uint32_t us)
374{
375 /* The u32TimeOutCnt value must be greater than the max delay time of 1398ms if HCLK=12MHz */
376 uint32_t u32TimeOutCnt = SystemCoreClock * 2;
377
378 SysTick->LOAD = us * CyclesPerUs;
379 SysTick->VAL = (0x00);
380 SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk | SysTick_CTRL_ENABLE_Msk;
381
382 /* Waiting for down-count to zero */
383 while((SysTick->CTRL & SysTick_CTRL_COUNTFLAG_Msk) == 0)
384 {
385 if(--u32TimeOutCnt == 0)
386 {
387 break;
388 }
389 }
390
391 /* Disable SysTick counter */
392 SysTick->CTRL = 0;
393
394 if(u32TimeOutCnt == 0)
395 return CLK_TIMEOUT_ERR;
396 else
397 return 0;
398}
