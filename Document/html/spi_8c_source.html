757 {

759 
760  if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_HXT)

764  else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_PCLK1)

769  if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK1SEL_Msk) == CLK_CLKSEL0_PCLK1SEL_HCLK_DIV2)
770  u32Freq = (u32HCLKFreq / 2);
771  else
772  u32Freq = u32HCLKFreq;
773  }
774  else
775  u32Freq = __HIRC; /* Clock source is HIRC */
776

807 uint32_t I2S_Open(SPI_T *i2s, uint32_t u32MasterSlave, uint32_t u32SampleRate, uint32_t u32WordWidth, uint32_t u32Channels, uint32_t u32DataFormat)
808 {
809  uint32_t u32Divider;
810  uint32_t u32BitRate, u32SrcClk;
811  uint32_t u32HCLKFreq;
812 
813  /* Reset SPI/I2S */
814  SYS->IPRST1 |= SYS_IPRST1_SPI1RST_Msk;
815  SYS->IPRST1 &= ~SYS_IPRST1_SPI1RST_Msk;
816 
817  /* Configure I2S controller */
818  i2s->I2SCTL = u32MasterSlave | u32WordWidth | u32Channels | u32DataFormat;
819  /* Set TX and RX FIFO threshold to middle value */
820  i2s->FIFOCTL = I2S_FIFO_TX_LEVEL_WORD_2 | I2S_FIFO_RX_LEVEL_WORD_2;
821 
822  if(u32MasterSlave == SPI_MASTER)
823  {
824  /* Get the source clock rate */
825  u32SrcClk = I2S_GetSourceClockFreq(i2s);
826 
827  /* Calculate the bit clock rate */
828  u32BitRate = u32SampleRate * ((u32WordWidth >> SPI_I2SCTL_WDWIDTH_Pos) + 1) * 16;
829  u32Divider = ((u32SrcClk / u32BitRate) >> 1) - 1;
830  /* Set BCLKDIV setting */
831  i2s->I2SCLK = (i2s->I2SCLK & ~SPI_I2SCLK_BCLKDIV_Msk) | (u32Divider << SPI_I2SCLK_BCLKDIV_Pos);
832 
833  /* Calculate bit clock rate */
834  u32BitRate = u32SrcClk / ((u32Divider + 1) * 2);
835  /* Calculate real sample rate */
836  u32SampleRate = u32BitRate / (((u32WordWidth >> SPI_I2SCTL_WDWIDTH_Pos) + 1) * 16);
837 
838  /* Enable TX function, RX function and I2S mode. */
839  i2s->I2SCTL |= (SPI_I2SCTL_RXEN_Msk | SPI_I2SCTL_TXEN_Msk | SPI_I2SCTL_I2SEN_Msk);
840 
841  /* Return the real sample rate */
842  return u32SampleRate;
843  }
844  else
845  {
846  /* Set BCLKDIV = 0 */
847  i2s->I2SCLK &= ~SPI_I2SCLK_BCLKDIV_Msk;
848  /* Get system clock frequency */
849  u32HCLKFreq = CLK_GetHCLKFreq();
850 
851  /* Set the peripheral clock rate to equal APB clock rate */
852  CLK->CLKSEL2 = (CLK->CLKSEL2 & (~CLK_CLKSEL2_SPI1SEL_Msk)) | CLK_CLKSEL2_SPI1SEL_PCLK1;
853  /* Enable TX function, RX function and I2S mode. */
854  i2s->I2SCTL |= (SPI_I2SCTL_RXEN_Msk | SPI_I2SCTL_TXEN_Msk | SPI_I2SCTL_I2SEN_Msk);
855  /* Return slave peripheral clock rate */
856  if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK1SEL_Msk) == CLK_CLKSEL0_PCLK1SEL_HCLK_DIV2)
857  return (u32HCLKFreq / 2);
858  else
859  return u32HCLKFreq;
860  }
861 }
