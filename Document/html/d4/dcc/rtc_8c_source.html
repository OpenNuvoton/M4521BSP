70int32_t RTC_Open(S_RTC_TIME_DATA_T *sPt)
71{
72 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
73
74 RTC->INIT = RTC_INIT_KEY;
75
76 if(RTC->INIT != RTC_INIT_ACTIVE_Msk)
77 {
78 RTC->INIT = RTC_INIT_KEY;
79 while(RTC->INIT != RTC_INIT_ACTIVE_Msk)
80 {
81 if(u32TimeOutCount == 0) return -1;
82 u32TimeOutCount--;
83 }
84 }
85
86 if(sPt == 0)
87 return 0;
88
89 /* Set RTC date and time */
90 RTC_SetDateAndTime(sPt);
91
92 /* Waiting for RTC settings stable */
93 u32TimeOutCount = SystemCoreClock;
94 while((RTC->RWEN & RTC_RWEN_RWENF_Msk) == RTC_RWEN_RWENF_Msk)
95 {
96 if(u32TimeOutCount == 0) return -1;
97 u32TimeOutCount--;
98 }
99
100 return 0;
101}

126void RTC_32KCalibration(int32_t i32FrequencyX100)
127{
128 int32_t i32RegInt, i32RegFra;
129
130 /* Compute integer and fraction for RTC FCR register */
131 i32RegInt = (i32FrequencyX100 / 100) - RTC_FCR_REFERENCE;
132 i32RegFra = (((i32FrequencyX100 % 100)) * 60) / 100;
133
134 /* Judge Integer part is reasonable */
135 if((i32RegInt < 0) | (i32RegInt > 15))
136 {
137 return ;
138 }
139
140 RTC_WaitAccessEnable();
141 RTC->FREQADJ = (uint32_t)((i32RegInt << 8) | i32RegFra);
142}

162void RTC_GetDateAndTime(S_RTC_TIME_DATA_T *sPt)
163{
164 uint32_t u32Tmp;
165
166 sPt->u32TimeScale = RTC->CLKFMT & RTC_CLKFMT_24HEN_Msk; /* 12/24-hour */
167 sPt->u32DayOfWeek = RTC->WEEKDAY & RTC_WEEKDAY_WEEKDAY_Msk; /* Day of the week */
168
169 /* Get [Date digit] data */
170 g_u32hiYear = (RTC->CAL & RTC_CAL_TENYEAR_Msk) >> RTC_CAL_TENYEAR_Pos;
171 g_u32loYear = (RTC->CAL & RTC_CAL_YEAR_Msk) >> RTC_CAL_YEAR_Pos;
172 g_u32hiMonth = (RTC->CAL & RTC_CAL_TENMON_Msk) >> RTC_CAL_TENMON_Pos;
173 g_u32loMonth = (RTC->CAL & RTC_CAL_MON_Msk) >> RTC_CAL_MON_Pos;
174 g_u32hiDay = (RTC->CAL & RTC_CAL_TENDAY_Msk) >> RTC_CAL_TENDAY_Pos;
175 g_u32loDay = (RTC->CAL & RTC_CAL_DAY_Msk) >> RTC_CAL_DAY_Pos;
176
177 /* Get [Time digit] data */
178 g_u32hiHour = (RTC->TIME & RTC_TIME_TENHR_Msk) >> RTC_TIME_TENHR_Pos;
179 g_u32loHour = (RTC->TIME & RTC_TIME_HR_Msk) >> RTC_TIME_HR_Pos;
180 g_u32hiMin = (RTC->TIME & RTC_TIME_TENMIN_Msk) >> RTC_TIME_TENMIN_Pos;
181 g_u32loMin = (RTC->TIME & RTC_TIME_MIN_Msk) >> RTC_TIME_MIN_Pos;
182 g_u32hiSec = (RTC->TIME & RTC_TIME_TENSEC_Msk) >> RTC_TIME_TENSEC_Pos;
183 g_u32loSec = (RTC->TIME & RTC_TIME_SEC_Msk) >> RTC_TIME_SEC_Pos;
184
185 /* Compute to 20XX year */
186 u32Tmp = (g_u32hiYear * 10);
187 u32Tmp += g_u32loYear;
188 sPt->u32Year = u32Tmp + RTC_YEAR2000;
189
190 /* Compute 0~12 month */
191 u32Tmp = (g_u32hiMonth * 10);
192 sPt->u32Month = u32Tmp + g_u32loMonth;
193
194 /* Compute 0~31 day */
195 u32Tmp = (g_u32hiDay * 10);
196 sPt->u32Day = u32Tmp + g_u32loDay;
197
198 /* Compute 12/24 hour */
199 if(sPt->u32TimeScale == RTC_CLOCK_12)
200 {
201 u32Tmp = (g_u32hiHour * 10);
202 u32Tmp += g_u32loHour;
203 sPt->u32Hour = u32Tmp; /* AM: 1~12. PM: 21~32. */
204
205 if(sPt->u32Hour >= 21)
206 {
207 sPt->u32AmPm = RTC_PM;
208 sPt->u32Hour -= 20;
209 }
210 else
211 {
212 sPt->u32AmPm = RTC_AM;
213 }
214
215 u32Tmp = (g_u32hiMin * 10);
216 u32Tmp += g_u32loMin;
217 sPt->u32Minute = u32Tmp;
218
219 u32Tmp = (g_u32hiSec * 10);
220 u32Tmp += g_u32loSec;
221 sPt->u32Second = u32Tmp;
222 }
223 else
224 {
225 u32Tmp = (g_u32hiHour * 10);
226 u32Tmp += g_u32loHour;
227 sPt->u32Hour = u32Tmp;
228
229 u32Tmp = (g_u32hiMin * 10);
230 u32Tmp += g_u32loMin;
231 sPt->u32Minute = u32Tmp;
232
233 u32Tmp = (g_u32hiSec * 10);

235 sPt->u32Second = u32Tmp;

357void RTC_SetDateAndTime(S_RTC_TIME_DATA_T *sPt)
358{
359 uint32_t u32RegCAL, u32RegTIME;
360
361 if(sPt == 0)
362 return ;
363
364 /*-----------------------------------------------------------------------------------------------------*/
365 /* Set RTC 24/12 hour setting and Day of the Week */
366 /*-----------------------------------------------------------------------------------------------------*/
367 RTC_WaitAccessEnable();
368 if(sPt->u32TimeScale == RTC_CLOCK_12)
369 {
370 RTC->CLKFMT &= ~RTC_CLKFMT_24HEN_Msk;
371
372 /*-------------------------------------------------------------------------------------------------*/
373 /* Important, range of 12-hour PM mode is 21 up to 32 */
374 /*-------------------------------------------------------------------------------------------------*/
375 if(sPt->u32AmPm == RTC_PM)
376 sPt->u32Hour += 20;
377 }
378 else
379 {
380 RTC->CLKFMT |= RTC_CLKFMT_24HEN_Msk;
381 }
382
383 /* Set Day of the Week */
384 RTC->WEEKDAY = sPt->u32DayOfWeek;
385
386 /*-----------------------------------------------------------------------------------------------------*/
387 /* Set RTC Current Date and Time */
388 /*-----------------------------------------------------------------------------------------------------*/
389 u32RegCAL = ((sPt->u32Year - RTC_YEAR2000) / 10) << 20;
390 u32RegCAL |= (((sPt->u32Year - RTC_YEAR2000) % 10) << 16);
391 u32RegCAL |= ((sPt->u32Month / 10) << 12);
392 u32RegCAL |= ((sPt->u32Month % 10) << 8);
393 u32RegCAL |= ((sPt->u32Day / 10) << 4);
394 u32RegCAL |= (sPt->u32Day % 10);
395
396 u32RegTIME = ((sPt->u32Hour / 10) << 20);
397 u32RegTIME |= ((sPt->u32Hour % 10) << 16);
398 u32RegTIME |= ((sPt->u32Minute / 10) << 12);
399 u32RegTIME |= ((sPt->u32Minute % 10) << 8);
400 u32RegTIME |= ((sPt->u32Second / 10) << 4);
401 u32RegTIME |= (sPt->u32Second % 10);
402
403 /*-----------------------------------------------------------------------------------------------------*/
404 /* Set RTC Calender and Time Loading */
405 /*-----------------------------------------------------------------------------------------------------*/
406 RTC_WaitAccessEnable();
407 RTC->CAL = (uint32_t)u32RegCAL;
408 RTC->TIME = (uint32_t)u32RegTIME;
409}
