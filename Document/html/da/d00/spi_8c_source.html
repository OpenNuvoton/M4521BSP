50{
51 uint32_t u32ClkSrc = 0, u32Div, u32HCLKFreq;
52
53 if(spi == SPI1)
54 /* Disable I2S mode */
55 spi->I2SCTL &= ~SPI_I2SCTL_I2SEN_Msk;
56
57 if(u32DataWidth == 32)
58 u32DataWidth = 0;
59
60 /* Get system clock frequency */
61 u32HCLKFreq = CLK_GetHCLKFreq();
62
63 if(u32MasterSlave == SPI_MASTER)
64 {
65 /* Default setting: slave selection signal is active low; disable automatic slave selection function. */
66 spi->SSCTL = SPI_SS_ACTIVE_LOW;
67
68 /* Default setting: MSB first, disable unit transfer interrupt, SP_CYCLE = 0. */
69 spi->CTL = u32MasterSlave | (u32DataWidth << SPI_CTL_DWIDTH_Pos) | (u32SPIMode) | SPI_CTL_SPIEN_Msk;
70
71 if(u32BusClock >= u32HCLKFreq)
72 {
73 /* Select PCLK as the clock source of SPI */
74 if(spi == SPI0)
75 CLK->CLKSEL2 = (CLK->CLKSEL2 & (~CLK_CLKSEL2_SPI0SEL_Msk)) | CLK_CLKSEL2_SPI0SEL_PCLK0;
76 else
77 CLK->CLKSEL2 = (CLK->CLKSEL2 & (~CLK_CLKSEL2_SPI1SEL_Msk)) | CLK_CLKSEL2_SPI1SEL_PCLK1;
78 }
79
80 /* Check clock source of SPI */
81 if(spi == SPI0)
82 {
83 if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI0SEL_Msk) == CLK_CLKSEL2_SPI0SEL_HXT)
84 u32ClkSrc = __HXT; /* Clock source is HXT */
85 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI0SEL_Msk) == CLK_CLKSEL2_SPI0SEL_PLL)
86 u32ClkSrc = CLK_GetPLLClockFreq(); /* Clock source is PLL */
87 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI0SEL_Msk) == CLK_CLKSEL2_SPI0SEL_PCLK0)
88 {
89 /* Clock source is PCLK0 */
90 if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK0SEL_Msk) == CLK_CLKSEL0_PCLK0SEL_HCLK_DIV2)
91 u32ClkSrc = (u32HCLKFreq / 2);
92 else
93 u32ClkSrc = u32HCLKFreq;
94 }
95 else
96 u32ClkSrc = __HIRC; /* Clock source is HIRC */
97 }
98 else
99 {
100 if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_HXT)
101 u32ClkSrc = __HXT; /* Clock source is HXT */
102 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_PLL)
103 u32ClkSrc = CLK_GetPLLClockFreq(); /* Clock source is PLL */
104 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_PCLK1)
105 {
106 /* Clock source is PCLK1 */
107 if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK1SEL_Msk) == CLK_CLKSEL0_PCLK1SEL_HCLK_DIV2)
108 u32ClkSrc = (u32HCLKFreq / 2);
109 else
110 u32ClkSrc = u32HCLKFreq;
111 }
112 else
113 u32ClkSrc = __HIRC; /* Clock source is HIRC */
114 }
115
116 if(u32BusClock >= u32HCLKFreq)
117 {
118 /* Set DIVIDER = 0 */
119 spi->CLKDIV = 0;
120 /* Return master peripheral clock rate */
121 return u32ClkSrc;
122 }
123 else if(u32BusClock >= u32ClkSrc)
124 {
125 /* Set DIVIDER = 0 */
126 spi->CLKDIV = 0;
127 /* Return master peripheral clock rate */
128 return u32ClkSrc;
129 }
130 else if(u32BusClock == 0)
131 {
132 /* Set DIVIDER to the maximum value 0xFF. f_spi = f_spi_clk_src / (DIVIDER + 1) */
133 spi->CLKDIV |= SPI_CLKDIV_DIVIDER_Msk;
134 /* Return master peripheral clock rate */
135 return (u32ClkSrc / (0xFF + 1));
136 }
137 else

140 if(u32Div > 0xFF)
141 {
142 u32Div = 0xFF;
143 spi->CLKDIV |= SPI_CLKDIV_DIVIDER_Msk;
144 /* Return master peripheral clock rate */
145 return (u32ClkSrc / (0xFF + 1));
146 }
147 else
148 {
149 spi->CLKDIV = (spi->CLKDIV & (~SPI_CLKDIV_DIVIDER_Msk)) | (u32Div << SPI_CLKDIV_DIVIDER_Pos);
150 /* Return master peripheral clock rate */
151 return (u32ClkSrc / (u32Div + 1));
152 }
153 }
154 }
155 else /* For slave mode, force the SPI peripheral clock rate to equal APB clock rate. */
156 {
157 /* Default setting: slave selection signal is low level active. */
158 spi->SSCTL = SPI_SS_ACTIVE_LOW;
159
160 /* Default setting: MSB first, disable unit transfer interrupt, SP_CYCLE = 0. */

170 /* Return slave peripheral clock rate */

172 return (u32HCLKFreq / 2);

757static uint32_t I2S_GetSourceClockFreq(SPI_T *i2s)
758{
759 uint32_t u32Freq, u32HCLKFreq;
760
761 if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_HXT)
762 u32Freq = __HXT; /* Clock source is HXT */
763 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_PLL)
764 u32Freq = CLK_GetPLLClockFreq(); /* Clock source is PLL */
765 else if((CLK->CLKSEL2 & CLK_CLKSEL2_SPI1SEL_Msk) == CLK_CLKSEL2_SPI1SEL_PCLK1)
766 {
767 /* Get system clock frequency */
768 u32HCLKFreq = CLK_GetHCLKFreq();
769 /* Clock source is PCLK1 */
770 if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK1SEL_Msk) == CLK_CLKSEL0_PCLK1SEL_HCLK_DIV2)
771 u32Freq = (u32HCLKFreq / 2);
772 else
773 u32Freq = u32HCLKFreq;
774 }
775 else
776 u32Freq = __HIRC; /* Clock source is HIRC */
777
778 return u32Freq;
779}

808uint32_t I2S_Open(SPI_T *i2s, uint32_t u32MasterSlave, uint32_t u32SampleRate, uint32_t u32WordWidth, uint32_t u32Channels, uint32_t u32DataFormat)
809{
810 uint32_t u32Divider;
811 uint32_t u32BitRate, u32SrcClk;
812 uint32_t u32HCLKFreq;
813
814 /* Reset SPI/I2S */
815 SYS->IPRST1 |= SYS_IPRST1_SPI1RST_Msk;
816 SYS->IPRST1 &= ~SYS_IPRST1_SPI1RST_Msk;
817
818 /* Configure I2S controller */
819 i2s->I2SCTL = u32MasterSlave | u32WordWidth | u32Channels | u32DataFormat;
820 /* Set TX and RX FIFO threshold to middle value */
821 i2s->FIFOCTL = I2S_FIFO_TX_LEVEL_WORD_2 | I2S_FIFO_RX_LEVEL_WORD_2;
822
823 if(u32MasterSlave == SPI_MASTER)
824 {
825 /* Get the source clock rate */
826 u32SrcClk = I2S_GetSourceClockFreq(i2s);
827
828 /* Calculate the bit clock rate */
829 u32BitRate = u32SampleRate * ((u32WordWidth >> SPI_I2SCTL_WDWIDTH_Pos) + 1) * 16;
830 u32Divider = ((u32SrcClk / u32BitRate) >> 1) - 1;
831 /* Set BCLKDIV setting */
832 i2s->I2SCLK = (i2s->I2SCLK & ~SPI_I2SCLK_BCLKDIV_Msk) | (u32Divider << SPI_I2SCLK_BCLKDIV_Pos);
833
834 /* Calculate bit clock rate */
835 u32BitRate = u32SrcClk / ((u32Divider + 1) * 2);
836 /* Calculate real sample rate */
837 u32SampleRate = u32BitRate / (((u32WordWidth >> SPI_I2SCTL_WDWIDTH_Pos) + 1) * 16);
838
839 /* Enable TX function, RX function and I2S mode. */
840 i2s->I2SCTL |= (SPI_I2SCTL_RXEN_Msk | SPI_I2SCTL_TXEN_Msk | SPI_I2SCTL_I2SEN_Msk);
841
842 /* Return the real sample rate */
843 return u32SampleRate;
844 }
845 else
846 {
847 /* Set BCLKDIV = 0 */
848 i2s->I2SCLK &= ~SPI_I2SCLK_BCLKDIV_Msk;
849 /* Get system clock frequency */
850 u32HCLKFreq = CLK_GetHCLKFreq();
851
852 /* Set the peripheral clock rate to equal APB clock rate */
853 CLK->CLKSEL2 = (CLK->CLKSEL2 & (~CLK_CLKSEL2_SPI1SEL_Msk)) | CLK_CLKSEL2_SPI1SEL_PCLK1;
854 /* Enable TX function, RX function and I2S mode. */
855 i2s->I2SCTL |= (SPI_I2SCTL_RXEN_Msk | SPI_I2SCTL_TXEN_Msk | SPI_I2SCTL_I2SEN_Msk);
856 /* Return slave peripheral clock rate */
857 if((CLK->CLKSEL0 & CLK_CLKSEL0_PCLK1SEL_Msk) == CLK_CLKSEL0_PCLK1SEL_HCLK_DIV2)
858 return (u32HCLKFreq / 2);
859 else
860 return u32HCLKFreq;
861 }

979uint32_t I2S_EnableMCLK(SPI_T *i2s, uint32_t u32BusClock)
980{
981 uint32_t u32Divider;
982 uint32_t u32SrcClk;
983
984 u32SrcClk = I2S_GetSourceClockFreq(i2s);
985 if(u32BusClock == u32SrcClk)
986 u32Divider = 0;
987 else
988 {
989// u32Divider = (u32SrcClk / u32BusClock) >> 1;
990 u32Divider = (((u32SrcClk * 10UL / u32BusClock) >> 1) + 5UL) / 10UL;
991 /* MCLKDIV is a 6-bit width configuration. The maximum value is 0x3F. */
992 if(u32Divider > 0x3F)
993 u32Divider = 0x3F;
994 }
995
996 /* Write u32Divider to MCLKDIV (SPI_I2SCLK[5:0]) */
997 i2s->I2SCLK = (i2s->I2SCLK & ~SPI_I2SCLK_MCLKDIV_Msk) | (u32Divider << SPI_I2SCLK_MCLKDIV_Pos);
998
999 /* Enable MCLK output */
1000 i2s->I2SCTL |= SPI_I2SCTL_MCLKEN_Msk;
1001
1002 if(u32Divider == 0)
1003 return u32SrcClk; /* If MCLKDIV=0, master clock rate is equal to the source clock rate. */
1004 else
1005 return ((u32SrcClk >> 1) / u32Divider); /* If MCLKDIV>0, master clock rate = source clock rate / (MCLKDIV * 2) */
1006}
