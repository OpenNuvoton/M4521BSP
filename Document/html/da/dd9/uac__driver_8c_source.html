105static int au_probe(USB_DEV_T *dev, USB_IF_DESC_T *ifd, const USB_DEV_ID_T *id)
106{
107 EP_INFO_T *ep_info;
108 int ifnum;
109 UAC_DEV_T *audev, *p;
110 int i;
111
112 if((ifd->bInterfaceSubClass != SUBCLS_AUDIOCONTROL) &&
113 (ifd->bInterfaceSubClass != SUBCLS_AUDIOSTREAMING))
114 {
115 printf("Does not support audio sub-class %x\n", ifd->bInterfaceSubClass);
116 return UAC_RET_DRV_NOT_SUPPORTED;
117 }
118
119
120 ifnum = ifd->bInterfaceNumber;
121
122 audev = find_as_deivce_by_udev(dev);
123 if(audev == NULL)
124 {
125 audev = alloc_as_device();
126 if(audev == NULL)
127 return USB_ERR_NODEV;
128 }
129
130 audev->udev = dev;
131 audev->next = NULL;
132
133 USBAS_DBGMSG("au_probe called for ifnum %d\n", ifnum);
134
135 if(ifd->bInterfaceSubClass == SUBCLS_AUDIOSTREAMING)
136 {
137 ep_info = NULL;
138 for(i = 0; i < dev->ep_list_cnt; i++)
139 {
140 if((dev->ep_list[i].ifnum == ifnum) &&
141 ((dev->ep_list[i].bmAttributes & USB_ENDPOINT_XFERTYPE_MASK) == USB_ENDPOINT_XFER_ISOC))
142 {
143 ep_info = &dev->ep_list[i];
144 USBAS_DBGMSG("Isochronous Endpoint 0x%x found.\n", ep_info->bEndpointAddress);
145 }
146 }
147 if(ep_info == NULL)
148 {
149 USBAS_DBGMSG("couldn't find isochronous endpoints\n");
150 return USB_ERR_NODEV;
151 }
152
153 if((ep_info->bEndpointAddress & USB_ENDPOINT_DIR_MASK) == USB_DIR_IN)
154 {
155 audev->ep_au_in = ep_info;
156 audev->au_in_ifnum = ifnum;
157 USBAS_DBGMSG("Audio in stream interface is %d\n", ifnum);
158 }
159 else
160 {
161 audev->ep_au_out = ep_info;
162 audev->au_out_ifnum = ifnum;
163 USBAS_DBGMSG("Audio out stream interface is %d\n", ifnum);
164 }
165 }
166 else
167 {
168 audev->ctrl_ifnum = ifnum;
169 USBAS_DBGMSG("Audio control interface is %d\n", ifnum);
170 }
171
172 /*
173 * Chaining newly found Audio Class device to end of Audio Class device list.
174 */
175 if(g_audev_list == NULL)
176 g_audev_list = audev;
177 else
178 {
179 for(p = g_audev_list; p->next != NULL; p = p->next)
180 ;
181 p->next = audev;
182 }
183
184 return uac_config_parser(audev);
185}

188static void au_disconnect(USB_DEV_T *dev)
189{
190 UAC_DEV_T *audev, *p;
191 int i;
192
193 USBAS_DBGMSG("Audio Class device disconnected!\n");
194
195 audev = find_as_deivce_by_udev(dev);
196 if(audev == NULL)
197 return;
198
199 for(i = 0; i < ISO_IN_URB_CNT; i++)
200 {
201 if(audev->urbin[i])
202 {
203 USBH_UnlinkUrb(audev->urbin[i]);
204 USBH_FreeUrb(audev->urbin[i]);
205 }
206 }
207
208 for(i = 0; i < ISO_OUT_URB_CNT; i++)
209 {
210 if(audev->urbout[i])
211 {
212 USBH_UnlinkUrb(audev->urbout[i]);
213 USBH_FreeUrb(audev->urbout[i]);
214 }
215 }
216
217 /*
218 * Remove this Audio Class device from device list.
219 */
220 if(audev == g_audev_list)
221 g_audev_list = g_audev_list->next;
222 else
223 {
224 for(p = g_audev_list; p != NULL; p = p->next)
225 {
226 if(p->next == audev)
227 {
228 p->next = audev->next;
229 break;
230 }
231 }
232 }
233 free_au_device(audev);
234}
