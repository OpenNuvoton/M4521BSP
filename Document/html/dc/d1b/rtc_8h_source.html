231static __INLINE void RTC_WaitAccessEnable(void)
232{
233 uint32_t u32TimeOutCount = SystemCoreClock; // 1 second timeout
234 uint32_t i = 0;
235
236 /* To wait RWENF bit is cleared and enable RWENF bit (Access Enable bit) again */
237 while((RTC->RWEN & RTC_RWEN_RWENF_Msk) == RTC_RWEN_RWENF_Msk)
238 {
239 i++;
240 if(i > u32TimeOutCount) break;
241 }
242
243 RTC->RWEN = RTC_WRITE_KEY;
244
245 i = 0;
246 /* To wait RWENF bit is set and user can access the protected-register of RTC from now on */
247 while((RTC->RWEN & RTC_RWEN_RWENF_Msk) == 0x0)
248 {
249 i++;
250 if(i > u32TimeOutCount) break;
251 }
252}
