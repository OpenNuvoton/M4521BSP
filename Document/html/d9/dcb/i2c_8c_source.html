40uint32_t I2C_Open(I2C_T *i2c, uint32_t u32BusClock)
41{
42 uint32_t u32Div;
43
44 u32Div = (uint32_t)(((SystemCoreClock * 10) / (u32BusClock * 4) + 5) / 10 - 1); /* Compute proper divider for I2C clock */
45 i2c->CLKDIV = u32Div;

47 /* Enable I2C */

49

550void I2C_SMBusTimeout(I2C_T *i2c, uint32_t ms, uint32_t u32Pclk)
551{
552 uint32_t u32Div, u32Pclk_kHz;
553
554 i2c->BUSCTL &= ~I2C_BUSCTL_TIDLE_Msk;
555
556 /* DIV4 disabled */
557 i2c->TOCTL &= ~I2C_TOCTL_TOCEN_Msk;
558 u32Pclk_kHz = u32Pclk / 1000;
559 u32Div = ((ms * u32Pclk_kHz) / (16 * 1024)) - 1;
560 if(u32Div <= 0xFF)
561 {
562 i2c->BUSTOUT = u32Div;
563 return;
564 }
565
566 /* DIV4 enabled */
567 i2c->TOCTL |= I2C_TOCTL_TOCEN_Msk;
568
569 i2c->BUSTOUT = (((ms * u32Pclk_kHz) / (16 * 1024 * 4)) - 1) & 0xFF; //The max value is 255
570}
